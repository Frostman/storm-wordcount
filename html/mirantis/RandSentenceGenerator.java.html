17   public class RandSentenceGenerator extends BaseRichSpout { 
18    
19       private SpoutOutputCollector collector; 
20       private Random random; 
21    
22       private String[] sentences;
 
23    
24       @Override 
25       public void open(Map map, TopologyContext ctx, SpoutOutputCollector collector) { 
26           this.collector = collector; 
27           this.random = new Random(); 
28           this.sentences = 
29                   ("test hello world example\n" + 
30                           " sample hello another\n" + 
31                           " twitter storm one framework" 
32                   ).split("\n"); 
33       }
 
34    
35       @Override 
36       public void nextTuple() { 
37           Utils.sleep(10); 
38           String sentence = sentences[random.nextInt(sentences.length)]; 
39           collector.emit(new Values(sentence)); 
40       }
 
41    
42       @Override 
43       public void declareOutputFields(OutputFieldsDeclarer declarer) { 
44           declarer.declare(new Fields("sentence")); 
45       }
