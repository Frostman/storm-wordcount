1    package mirantis;

3    import backtype.storm.Config; 
4    import backtype.storm.LocalCluster; 
5    import backtype.storm.StormSubmitter; 
6    import backtype.storm.topology.TopologyBuilder;

8     
9    /** 
10    * @author slukjanov 
11    */ 
12   public class WordCounter { 
13    
14       public static void main(String[] args) throws Exception { 
15           TopologyBuilder builder = new TopologyBuilder(); 
16    
17           builder.setSpout("source", new RandSentenceGenerator(), 3); 
18           builder.setSpout("ping", new PingSpout()); 
19    
20           builder.setBolt("split", new SplitSentence(), 8) 
21                   .shuffleGrouping("source"); 
22           builder.setBolt("count", new WordCount(), 12) 
23                   .fieldsGrouping("split", new Fields("word")) 
24                   .allGrouping("ping"); 
25    
26           Config conf = new Config(); 
27           conf.setDebug(true); 
28    
29           if (args != null && args.length > 0) { 
30               conf.setNumWorkers(3); 
31               if (args.length > 1) { 
32                   conf.setNumWorkers(Integer.parseInt(args[1])); 
33               } 
34    
35               StormSubmitter.submitTopology(args[0], conf, builder.createTopology()); 
36           } else { 
37               conf.setMaxTaskParallelism(3); 
38    
39               LocalCluster cluster = new LocalCluster(); 
40               cluster.submitTopology("word-count", conf, builder.createTopology()); 
41    
42               Thread.sleep(30000); 
43    
44               cluster.shutdown(); 
45           } 
46       } 
47    
48   }
