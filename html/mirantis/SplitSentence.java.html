13   public class SplitSentence extends BaseBasicBolt { 
14    
15       @Override 
16       public void execute(Tuple tuple, BasicOutputCollector collector) { 
17           String sentence = tuple.getString(0); 
18           for (String word : sentence.split("\\s")) { 
19               collector.emit(new Values(word)); 
20           } 
21       }
 
22
