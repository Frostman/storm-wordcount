9    import com.google.common.collect.HashMultiset; 
10   import org.apache.log4j.Logger; 
11    
12   import java.util.Map;

14   /** 
15    * @author slukjanov

17   public class WordCount extends BaseBasicBolt { 
18    
19       private Logger logger; 
20       private String name; 
21       private int task; 
22    
23       private HashMultiset<String> words = HashMultiset.create();
 
24    
25       @Override 
26       public void prepare(Map conf, TopologyContext ctx) { 
27           super.prepare(conf, ctx); 
28           logger = Logger.getLogger(this.getClass()); 
29           name = ctx.getThisComponentId(); 
30           task = ctx.getThisTaskIndex(); 
31       }
 
32    
33       @Override 
34       public void execute(Tuple tuple, BasicOutputCollector collector) { 
35           String source = tuple.getSourceComponent(); 
36           if ("split".equals(source)) { 
37               words.add(tuple.getString(0)); 
38           } else if ("ping".equals(source)) { 
39               logger.warn("RESULT " + name + ":" + task + " :: " + words); 
40           } 
41       }
 
42    
43       @Override 
44       public void declareOutputFields(OutputFieldsDeclarer declarer) { 
45           declarer.declare(new Fields("word", "count")); 
46       } 
47    
48   }
